-- Only controls and OSD are rotated on Video output.

-- 
-- Keyboard inputs :
--
-- Fire = Fire or Space
-- Thrust = Fire2 or ALT
-- Smart Bomb = Fire3 or CTRL
-- Hyperspace = Fire4 or 1
-- Change Direction = Left or Right
-- Up = Up
-- Down = Down

-- Advance = A
-- Auto up = U      
-- Score_reset = H  
--
-- Joystick support.
-- 
---------------------------------------------------------------------------------


DEFENDER.ROM is required at the root of the SD-Card.
